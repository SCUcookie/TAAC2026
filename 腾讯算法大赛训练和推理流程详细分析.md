### 5. 训练循环详细流程

#### 单个训练步骤
```python
for step, batch in train_loader:
    # 1. 数据解包
    seq, pos, neg, token_type, next_token_type, next_action_type, seq_feat, pos_feat, neg_feat = batch
    
    # 2. 数据移到GPU
    seq = seq.to(args.device)
    pos = pos.to(args.device)  
    neg = neg.to(args.device)
    
    # 3. 前向传播
    if args.loss_type == 'bce':
        # BCE损失计算
        pos_logits, neg_logits, loss_mask = model(...)
        loss = bce_criterion(pos_logits[indices], pos_labels[indices])
        loss += bce_criterion(neg_logits[indices], neg_labels[indices])
        
    elif args.loss_type == 'infonce':
        # InfoNCE对比学习损失
        seq_embs = compute_embeddings(model, seq, seq_feat, token_type)
        pos_embs = model.feat2emb(pos, pos_feat, include_user=False)
        neg_embs = model.feat2emb(neg, neg_feat, include_user=False)
        loss = model.compute_infonce_loss(seq_embs, pos_embs, neg_embs, loss_mask)
        
    elif args.loss_type == 'mix':
        # 混合损失：BCE + InfoNCE
        bce_loss = compute_bce_loss(...)
        infonce_loss = compute_infonce_loss(...)
        loss = bce_loss  # 当前实现只使用BCE部分
    
    # 4. L2正则化
    for param in model.item_emb.parameters():
        loss += args.l2_emb * torch.norm(param)
    
    # 5. 反向传播
    optimizer.zero_grad()
    loss.backward()
    
    # 6. 梯度裁剪
    torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
    
    # 7. 学习率调度 (前800步warmup)
    if global_step < args.warmup_steps:
        lr_scalar = min(1.0, float(global_step + 1) / args.warmup_steps)
        for pg in optimizer.param_groups:
            pg["lr"] = lr_scalar * args.lr
    
    # 8. 参数更新
    optimizer.step()
```

```python
def process_cold_start_feat(feat):
    """处理冷启动特征，字符串类型转换为0"""
    processed_feat = {}
    for feat_id, feat_value in feat.items():
        if type(feat_value) == str:
            processed_feat[feat_id] = 0
        elif type(feat_value) == list:
            value_list = [0 if type(v) == str else v for v in feat_value]
            processed_feat[feat_id] = value_list
        else:
            processed_feat[feat_id] = feat_value
    return processed_feat
```

```python
# Focal Loss处理类别不平衡 (9.727% CTR)
class FocalLoss(nn.Module):
    def __init__(self, alpha=0.25, gamma=2.0):
        super().__init__()
        self.alpha = alpha
        self.gamma = gamma
    
    def forward(self, inputs, targets):
        ce_loss = F.cross_entropy(inputs, targets, reduction='none')
        pt = torch.exp(-ce_loss)
        focal_loss = self.alpha * (1-pt)**self.gamma * ce_loss
        return focal_loss.mean()
```

```python
class AdversarialTraining:
    def adversarial_step(self, features, labels):
        # 生成对抗样本
        features.requires_grad = True
        loss = self.model.compute_loss(features, labels)
        grad = torch.autograd.grad(loss, features)[0]
        
        # 添加对抗扰动
        adversarial_features = features + self.epsilon * grad.sign()
        adv_loss = self.model.compute_loss(adversarial_features, labels)
        
        return adv_loss
```

```python
def stratified_validation(predictions, labels, user_activity_levels):
    """按用户活跃度分层评估"""
    results = {}
    activity_weights = {'inactive': 0.017, 'low': 0.326, 'medium': 0.609, 'high': 0.048}
    
    weighted_auc = 0
    for level in ['inactive', 'low', 'medium', 'high']:
        mask = (user_activity_levels == level)
        if mask.sum() > 0:
            level_auc = roc_auc_score(labels[mask], predictions[mask])
            results[f'auc_{level}'] = level_auc
            weighted_auc += level_auc * activity_weights[level]
    
    results['weighted_auc'] = weighted_auc
    return results
```

```python
import torch.jit

class ModelServer:
    def __init__(self, model_path, device='cuda'):
        # 加载JIT编译的模型
        self.model = torch.jit.load(model_path)
        self.model.eval()
        self.device = device
        
    def predict(self, user_sequences, features):
        with torch.no_grad():
            sequences = torch.tensor(user_sequences).to(self.device)
            features = {k: torch.tensor(v).to(self.device) 
                       for k, v in features.items()}
            
            predictions = self.model(sequences, features)
            return predictions.cpu().numpy()
```
